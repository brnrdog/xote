    <p>


    <h3 id="writing-components"> {"Writing Components"} </h3>
    <h4 id="component-module-pattern"> {"Recommended Pattern"} </h4>
    <p>
      {"Use a module with a "}
      <code> {"make"} </code>
      {" function and annotate it with "}
      <code> {"@jsx.component"} </code>
      {". ReScript derives the props shape from labeled arguments."}
    </p>
    
```rescript
open Xote

module Greeting = {
  @jsx.component
  let make = (~name: string, ~emphasis=false) => {
    <div class={emphasis ? "greeting strong" : "greeting"}>
      <h1> {View.text(`Hello, ${name}`)} </h1>
    </div>
  }
}

let app = () => {
  <Greeting name="World" emphasis />
}
```

    <h4 id="function-api"> {"Function API"} </h4>
    <p>
      {"The lower-level API is still useful when JSX is not a good fit."}
    </p>
    
```rescript
open Xote

let greeting = (name: string) => {
  Html.div(
    ~children=[
      Html.h1(~children=[View.text(`Hello, ${name}`)], ())
    ],
    (),
  )
}
```

    <h3 id="reactive-output"> {"Reactive Output"} </h3>
    <p>
      {"JSX expressions are just nodes. For reactive text, use "}
      <code> {"View.signalText"} </code>
      {". For arrays and lists, prefer "}
      <code> {"View.eachWithKey"} </code>
      {" when item identity matters."}
    </p>
    
```rescript
let count = Signal.make(0)

<div>
  {View.signalText(() => `Count: ${Signal.get(count)->Int.toString}`)}
</div>
```

    <h3 id="attributes-and-events"> {"Attributes and Events"} </h3>
    <p>
      {"In JSX, common HTML props are exposed directly. In the function API, use "}
      <code> {"View.Attr.string"} </code>
      {", "}
      <code> {"View.Attr.signal"} </code>
      {", and "}
      <code> {"View.Attr.compute"} </code>
      {"."}
    </p>
    
```rescript
let isActive = Signal.make(false)

let toggle = (_evt: Dom.event) => {
  Signal.update(isActive, active => !active)
}

<button
  class={Signal.get(isActive) ? "btn active" : "btn"}
  onClick={toggle}>
  {View.text("Toggle")}
</button>
```

    
```rescript
Html.button(
  ~attrs=[
    View.Attr.compute("class", () =>
      Signal.get(isActive) ? "btn active" : "btn"
    ),
  ],
  ~events=[("click", toggle)],
  ~children=[View.text("Toggle")],
  (),
)
```

    <p>
      {"In JSX, use "}
      <code> {"class"} </code>
      {", not "}
      <code> {"className"} </code>
      {". Use "}
      <code> {"type_"} </code>
      {" for the HTML "}
      <code> {"type"} </code>
      {" attribute because "}
      <code> {"type"} </code>
      {" is reserved in ReScript."}
    </p>

    <h3 id="lists"> {"Lists"} </h3>
    <p>
      {"Use "}
      <code> {"View.each"} </code>
      {" for simple arrays that can be fully re-rendered, and "}
      <code> {"View.eachWithKey"} </code>
      {" when item identity matters."}
    </p>
    
```rescript
let items = Signal.make(["Apple", "Banana", "Cherry"])

<ul>
  {View.each(items, item => <li> {View.text(item)} </li>)}
</ul>
```

    
```rescript
type todo = {id: string, text: string}
let todos = Signal.make([
  {id: "1", text: "Write docs"},
  {id: "2", text: "Ship release"},
])

<ul>
  {View.eachWithKey(
    todos,
    todo => todo.id,
    todo => <li> {View.text(todo.text)} </li>,
  )}
</ul>
```

    <p>
      {"Choose stable keys. Database IDs and route slugs are good. Array indexes are not."}
    </p>

    <h3 id="mounting"> {"Mounting"} </h3>
    <p>
      {"Use "}
      <code> {"View.mount"} </code>
      {" when you already have a DOM element, or "}
      <code> {"View.mountById"} </code>
      {" when you want to look one up by id."}
    </p>
    
```rescript
let app = () => {

View.mountById(app(), "app")
```

    <h3 id="best-practices"> {"Best Practices"} </h3>
    <ul>
      <li>
        {"Default to the JSX module pattern when there is no reason to drop lower."}
      </li>
      <li>
        {"Keep state close to where it is used. Local signals are cheap and usually easier to follow."}
      </li>
      <li>
        {"Use "}
        <code> {"View.eachWithKey"} </code>
        {" for collections that reorder, insert, or preserve local DOM state."}
      </li>
      <li>
        {"Be explicit about reactive output so the update boundaries stay readable in the component."}
      </li>
    </ul>
